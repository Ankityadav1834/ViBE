"""
================================================================================
  COMBINED DIS + SEI STRESS MODEL  —  C-rate → Stress → Capacity Fade
================================================================================
  Physics Layer 1: Diffusion-Induced Stress (DIS)
    σ_r, σ_θ, σ_h, σ_vm throughout particle volume
    Timescale: diffusion time τ = R_s²/D_s  (seconds–hours, within one cycle)
    Reference: Christensen & Newman (2006)

  Physics Layer 2: SEI Mismatch Stress
    Biaxial stress in thin SEI film at graphite surface
    σ_SEI = E_SEI_b * (Ω/3) * (c_surf - c_surf_ref)
    Timescale: cycle-to-cycle accumulation (days–months)
    Reference: Safari & Delacourt (2011), Reniers et al. (2019)

  Timescale Problem & Solution:
    τ_DIS  ~ 2564 s   (within one cycle, resolved at dt=12s)
    τ_SEI  ~ 10^6 cycles (slow accumulation)
    Solution: OPERATOR SPLITTING
      Fast loop: DIS computed at dt=12s within each cycle
      Slow loop: SEI thickness updated once per cycle
      Coupling:  DIS surface stress → SEI cracking → enhanced SEI growth
                 SEI thickness → additional impedance → modified concentration BC

  Stress Superposition (at graphite particle surface):
    σ_total_hoop = σ_θ_DIS(r=R) + σ_SEI_mismatch
    These are additive because both are tangential at r=R.
    DIS σ_r(R) = 0 by boundary condition; SEI does not change this.

  SEI Capacity Fade (two mechanisms):
    (A) Li consumption: ΔQ/Q = 2F*(ρ_SEI/M_SEI)*Δδ*a_s / (F*c_max*ε_s)
    (B) Stress-enhanced cracking: σ_SEI > σ_crack → fresh surface → more growth

  Feedback Loop (stress → more SEI → more fade):
    Higher C-rate → larger DIS → larger σ_SEI_peak → more cracking
    → more SEI growth per cycle → faster capacity fade

================================================================================
"""

            c_max = 30555.0,     # mol/m³
            c0_soc= 0.8,         # fully charged (ready to discharge)
            D_s   = 3.9e-14,     # m²/s
            R_s   = 10e-6,       # m

            sigma_f = 280e6,     # Pa fracture strength (Zhao et al. 2011)
        )

            E         = 10e9,      # Pa  (Christensen & Newman 2004)

            delta_0   = 5e-9,      # m   initial SEI thickness

        )
        # ── Cell geometry ────────────────────────────────────────────────
        self.cell = dict(
            I_1C  = 1.0,         # A/m² (normalised; ratios are exact)

        )


# ─────────────────────────────────────────────────────────────────────────────
# FVM SPHERICAL DIFFUSION SOLVER
# ─────────────────────────────────────────────────────────────────────────────
class SphericalDiffusionSolver:
    """Implicit-Euler FVM for ∂c/∂t = (D/r²)∂/∂r(r²∂c/∂r)"""

    def __init__(self, params, Nr=40):
        self.p   = params
        self.Nr  = Nr
        self.r   = np.linspace(0.0, float(params['R_s']), Nr+1)
        self.rc  = 0.5*(self.r[:-1]+self.r[1:])
        self.dr  = self.r[1]-self.r[0]
        self.c   = np.full(Nr, float(params['c0_soc'])*float(params['c_max']))

    def reset(self):
        self.c = np.full(self.Nr, float(self.p['c0_soc'])*float(self.p['c_max']))

    def _A(self):
        Nr, D, dr = self.Nr, float(self.p['D_s']), self.dr
        A = np.zeros((Nr,Nr))
        for i in range(Nr):
            rw,re = self.r[i],self.r[i+1]
            vol = (4/3)*np.pi*(re**3-rw**3)
            if i>0:
                cf = D*4*np.pi*rw**2/(dr*vol); A[i,i-1]+=cf; A[i,i]-=cf
            if i<Nr-1:
                cf = D*4*np.pi*re**2/(dr*vol); A[i,i+1]+=cf; A[i,i]-=cf
        return A

    def step(self, dt, j_n):
        Nr  = self.Nr
        M   = np.eye(Nr)-dt*self._A()
        rhs = self.c.copy()
        vol_N = (4/3)*np.pi*(self.r[-1]**3-self.r[-2]**3)
        rhs[-1] += dt*j_n*4*np.pi*float(self.p['R_s'])**2/vol_N
        self.c  = np.linalg.solve(M, rhs)
        self.c  = np.clip(self.c, 0.0, float(self.p['c_max']))
        return self.c.copy()

    def c_surface(self):
        return float(self.c[-1])

    def c_avg(self):
        v = (4/3)*np.pi*(self.r[1:]**3-self.r[:-1]**3)
        return float(np.dot(self.c,v)/((4/3)*np.pi*float(self.p['R_s'])**3))


# ─────────────────────────────────────────────────────────────────────────────
# RADIAL AVERAGE  (outer face — avoids r=0 singularity)
# ─────────────────────────────────────────────────────────────────────────────
def radial_avg(c, r_faces):
    """c̄(r) = (3/r_e³)∫₀^{r_e} c r'² dr' using outer face radius r_faces[i+1]."""
    v = (4/3)*np.pi*(r_faces[1:]**3-r_faces[:-1]**3)
    cb = np.zeros(len(c)); cv=cv2=0.0
    for i in range(len(c)):
        cv+=v[i]; cv2+=c[i]*v[i]
        cb[i] = cv2/((4/3)*np.pi*r_faces[i+1]**3)
    return cb


# ─────────────────────────────────────────────────────────────────────────────
# DIS STRESS CALCULATOR
# ─────────────────────────────────────────────────────────────────────────────
class DiffusionInducedStress:
    """
    Christensen & Newman (2006) Eqs. 17-18:
      σ_r  = (2EΩ/3(1-ν)) * [c̄ - c̄(r)]
      σ_θ  = (EΩ/3(1-ν))  * [2c̄ + c̄(r) - 3c(r)]
    BCs: σ_r(R)=0 (traction-free surface), σ_r(0)=σ_θ(0) (centre symmetry)
    """
    def __init__(self, params):
        E, nu, Om = float(params['E']), float(params['nu']), float(params['Omega'])
        self.pf   = E*Om/(3*(1-nu))
        self.sf   = float(params['sigma_f'])
        self.r_faces = np.linspace(0, float(params['R_s']), 41)

    def compute(self, c, c_bar):
        cb_r     = radial_avg(c, self.r_faces)
        pf       = self.pf
        sr       = 2*pf*(c_bar - cb_r)
        sth      =   pf*(2*c_bar + cb_r - 3*c)
        sh       = (sr + 2*sth)/3
        svm      = np.abs(sr - sth)
        return dict(sr=sr, sth=sth, sh=sh, svm=svm,
                    sth_surf=float(sth[-1]),  # surface hoop stress [Pa]
                    svm_peak=float(svm.max()),
                    FI=float(svm.max()/self.sf))


# ─────────────────────────────────────────────────────────────────────────────
# SEI LAYER MODEL
# ─────────────────────────────────────────────────────────────────────────────
class SEILayer:
    """
    SEI growth and stress model for one graphite electrode.

    Growth law (parabolic + stress-enhanced linear, Safari & Delacourt 2011):
      δ(n) = δ_0 + A_sei*sqrt(n) + B_sei*σ_DIS_peak*n

    Mismatch stress (biaxial thin film on spherical substrate):
      σ_SEI = E_SEI_b * (Ω/3) * (c_surf(t) - c_surf_ref)
      where c_surf_ref = concentration when SEI was last deposited (≈ formation SOC)
      Sign: tensile when graphite expands above reference (during charging)
            compressive when graphite shrinks (during discharging)

    SEI cracking criterion (Griffith-type for thin film):
      Cracking occurs when σ_SEI_tensile > K_Ic / sqrt(π*δ)
      → Stress-enhanced growth activated

    Capacity fade via Li consumption:
      ΔQ/Q_cell = 2F*(ρ/M)*Δδ*a_s / (F*c_max*ε_s)

    Timescale note:
      This class is updated ONCE PER CYCLE (slow loop).
      DIS stress is computed within each cycle (fast loop).
      Coupling: cycle-peak DIS surface stress → B_sei term here.
    """

    def __init__(self, mat: MaterialParameters):
        sp = mat.sei
        np_ = mat.neg

        self.delta       = float(sp['delta_0'])   # current SEI thickness [m]
        self.A_sei       = float(sp['A_sei'])
        self.B_sei       = float(sp['B_sei'])
        self.E_sei_b     = float(sp['E']) / (1-float(sp['nu']))  # biaxial modulus
        self.K_Ic        = float(sp['K_Ic'])
        self.M_sei       = float(sp['M_sei'])
        self.rho_sei     = float(sp['rho_sei'])
        self.sigma_intr  = float(sp['sigma_intrinsic'])

        self.Omega_n     = float(np_['Omega'])
        self.c_max       = float(np_['c_max'])
        self.c_surf_ref  = float(np_['c0_soc']) * float(np_['c_max'])  # formation SOC
        self.a_s         = 3*float(np_['eps_s'])/float(np_['R_s'])
        self.eps_s       = float(np_['eps_s'])
        self.F           = float(mat.cell['F'])

        # History tracking
        self.delta_hist        = [self.delta]
        self.sigma_mismatch_hist = [0.0]
        self.sigma_total_surf_hist = [0.0]
        self.cap_loss_SEI_hist  = [0.0]
        self.cracked_hist       = [False]
        self.cumulative_cap_loss = 0.0

    def mismatch_stress(self, c_surf: float) -> float:
        """
        σ_SEI = E_SEI_b * (Ω/3) * (c_surf - c_surf_ref)
        + σ_intrinsic (compressive growth stress, always present)

        Sign convention:
          c_surf > c_surf_ref → graphite expanded above reference → tensile in SEI
          c_surf < c_surf_ref → graphite contracted → compressive in SEI
        """
        delta_c    = c_surf - self.c_surf_ref
        sigma_mism = self.E_sei_b * (self.Omega_n/3.0) * delta_c
        return sigma_mism + self.sigma_intr  # total SEI stress

    def cracking_threshold(self) -> float:
        """
        Griffith-type cracking in thin film:
          σ_crack = K_Ic / sqrt(π * δ)
        Larger SEI → lower cracking threshold (easier to crack)
        This is physically correct: thicker brittle films crack more easily.
        """
        return self.K_Ic / np.sqrt(np.pi * max(self.delta, 1e-10))

    def step(self, cycle: int, sigma_DIS_peak: float,
             c_surf_peak_tensile: float) -> dict:
        """
        Advance SEI by one cycle.

        Parameters
        ----------
        cycle               : current cycle number (1-indexed)
        sigma_DIS_peak      : peak DIS von Mises stress during this cycle [Pa]
                              (drives stress-enhanced growth via cracking)
        c_surf_peak_tensile : peak surface concentration during CHARGING phase [mol/m³]
                              (the concentration that maximises tensile SEI stress)

        Returns
        -------
        dict with: delta_new, sigma_SEI, cracked, cap_loss_this_cycle
        """
        # 1. Compute current mismatch stress (at peak charging concentration)
        sigma_SEI = self.mismatch_stress(c_surf_peak_tensile)

        # 2. Check cracking
        sigma_crack_threshold = self.cracking_threshold()
        cracked = (sigma_SEI > sigma_crack_threshold)

        # 3. SEI growth this cycle
        #    Parabolic term: Δδ_parabolic = A*[sqrt(n)-sqrt(n-1)] ~ A/(2*sqrt(n))
        sqrt_n     = np.sqrt(float(cycle))
        sqrt_nm1   = np.sqrt(max(float(cycle)-1, 0.1))
        delta_parabolic = self.A_sei * (sqrt_n - sqrt_nm1)

        #    Stress-enhanced linear term (only when cracked or high DIS stress)
        #    Physical: cracks expose fresh graphite → faster SEI formation
        stress_enhancement = 1.0 + (3.0 if cracked else 0.0)   # 3× faster if cracked
        delta_stress = self.B_sei * sigma_DIS_peak * stress_enhancement

        delta_growth = delta_parabolic + delta_stress
        self.delta  += delta_growth

        # 4. Capacity loss from Li consumption this cycle
        #    ΔQ/Q = 2F*(ρ/M)*Δδ*a_s / (F*c_max*ε_s)
        #    Factor 2: each SEI unit cell consumes 2 Li⁺ (for Li₂CO₃)
        delta_Q_fraction = (2*self.F*(self.rho_sei/self.M_sei)*delta_growth*self.a_s
                            / (self.F*self.c_max*self.eps_s))
        self.cumulative_cap_loss += delta_Q_fraction

        # 5. Append to history
        self.delta_hist.append(self.delta)
        self.sigma_mismatch_hist.append(sigma_SEI)
        self.cracked_hist.append(cracked)
        self.cap_loss_SEI_hist.append(self.cumulative_cap_loss)

        return dict(
            delta          = self.delta,
            delta_growth   = delta_growth,
            sigma_SEI      = sigma_SEI,
            sigma_crack_th = sigma_crack_threshold,
            cracked        = cracked,
            cap_loss_frac  = self.cumulative_cap_loss,
        )

    def total_surface_stress(self, sigma_DIS_sth_surface: float,
                             c_surf: float) -> float:
        """
        Total tangential (hoop) stress at particle surface:
          σ_total = σ_θ_DIS(r=R) + σ_SEI_mismatch

        Physical basis:
          DIS σ_θ(R) acts in the tangential direction of the particle.
          SEI mismatch stress also acts tangentially (biaxial in-plane of film).
          Both drive surface cracking, so they add.
          DIS σ_r(R) = 0 by BC; SEI does not change the radial BC.
        """
        return sigma_DIS_sth_surface + self.mismatch_stress(c_surf)


# ─────────────────────────────────────────────────────────────────────────────
# ONE-CYCLE DIS SIMULATION  (fast inner loop)
# ─────────────────────────────────────────────────────────────────────────────
def simulate_one_cycle_dis(mat: MaterialParameters, C_rate: float,
                            Nt: int = Nt_PER_CYCLE) -> dict:
    """
    Run one complete discharge half-cycle at C_rate.
    Returns peak and time-series DIS stress quantities needed by SEI model.

    Timescale: t_cycle = t_1C / C_rate
    """
    p   = mat.cell
    F_  = float(p['F'])
    I   = C_rate * float(p['I_1C'])
    t_d = float(p['t_1C']) / C_rate
    dt  = t_d / Nt

    sol = SphericalDiffusionSolver(mat.neg, Nr=40)
    sol.reset()
    r_faces = sol.r
    dis     = DiffusionInducedStress(mat.neg)

    # Flux: j_neg < 0 (Li leaving graphite on discharge)
    a_s = 3*float(mat.neg['eps_s'])/float(mat.neg['R_s'])
    jn  = -I/(F_*a_s*float(p['L_neg']))

    # Track peak quantities for SEI model
    svm_peak    = 0.0
    sth_surf_max = -np.inf
    sth_surf_min =  np.inf
    c_surf_min   = float(mat.neg['c_max'])  # min surface conc (end of discharge)
    c_surf_max   = 0.0

    for _ in range(Nt):
        c = sol.step(dt, jn)
        c_bar = sol.c_avg()
        st = dis.compute(c, c_bar)
        svm_peak      = max(svm_peak, st['svm_peak'])
        sth_surf_max  = max(sth_surf_max, st['sth_surf'])
        sth_surf_min  = min(sth_surf_min, st['sth_surf'])
        c_surf_min    = min(c_surf_min, sol.c_surface())
        c_surf_max    = max(c_surf_max, sol.c_surface())

    return dict(
        sigma_vm_peak     = svm_peak,
        sigma_th_surf_max = sth_surf_max,
        sigma_th_surf_min = sth_surf_min,
        c_surf_end_discharge = sol.c_surface(),
        c_surf_min        = c_surf_min,
    )


# ─────────────────────────────────────────────────────────────────────────────
# MAIN SIMULATION: Operator-split DIS + SEI over many cycles
# ─────────────────────────────────────────────────────────────────────────────
def simulate_crate(mat: MaterialParameters, C_rate: float,
                   n_cycles: int = N_CYCLES) -> dict:
    """
    Operator-split simulation over n_cycles:

      For each cycle n:
        [Fast loop]  Run DIS simulation → extract peak stresses
        [Slow loop]  Advance SEI by one cycle using DIS stresses as input
        [Coupling]   SEI capacity loss → reduce available Li → reduced capacity

    Returns history of: capacity, SEI thickness, SEI stress, DIS stress,
    cracking events — all as arrays of length n_cycles.
    """
    sei = SEILayer(mat)
    c_max = float(mat.neg['c_max'])
    c0    = float(mat.neg['c0_soc'])

    # During charging, surface concentration returns to approximately c0_soc
    # (assumption: full charge each cycle)
    c_surf_ref_charging = c0 * c_max   # concentration at top of charge

    hist = dict(
        cycle           = [],
        capacity_retention = [],  # fraction of initial capacity
        sei_thickness   = [],     # m
        sei_stress      = [],     # Pa (mismatch stress at peak charging)
        sei_crack_th    = [],     # Pa (cracking threshold)
        cracked         = [],     # bool
        dis_vm_peak     = [],     # Pa
        cap_loss_SEI    = [],     # cumulative SEI Li loss fraction
        total_surf_stress = [],   # Pa (DIS hoop + SEI mismatch)
    )

    print(f"  {C_rate}C: simulating {n_cycles} cycles ...")

    for n in range(1, n_cycles+1):
        # ── Fast loop: one discharge cycle ──────────────────────────────
        dis_result = simulate_one_cycle_dis(mat, C_rate)

        # ── Slow loop: SEI growth for this cycle ─────────────────────────
        # During charging (not simulated in DIS, assumed full charge):
        # surface concentration goes back up to c_surf_ref_charging
        # This gives the tensile SEI stress phase
        sei_result = sei.step(
            cycle               = n,
            sigma_DIS_peak      = dis_result['sigma_vm_peak'],
            c_surf_peak_tensile = c_surf_ref_charging,  # peak tensile during charge
        )

        # ── Capacity: base DIS capacity minus SEI Li loss ─────────────────
        # Base capacity = 1.0 (normalised)
        # SEI Li loss reduces cyclable Li inventory
        cap_retention = max(0.70, 1.0 - sei_result['cap_loss_frac'])

        # Total surface stress (DIS hoop + SEI)
        total_surf = sei.total_surface_stress(
            dis_result['sigma_th_surf_max'],
            c_surf_ref_charging
        )

        hist['cycle'].append(n)
        hist['capacity_retention'].append(cap_retention)
        hist['sei_thickness'].append(sei_result['delta'])
        hist['sei_stress'].append(sei_result['sigma_SEI'])
        hist['sei_crack_th'].append(sei_result['sigma_crack_th'])
        hist['cracked'].append(sei_result['cracked'])
        hist['dis_vm_peak'].append(dis_result['sigma_vm_peak'])
        hist['cap_loss_SEI'].append(sei_result['cap_loss_frac'])
        hist['total_surf_stress'].append(total_surf)

    for k in hist:
        hist[k] = np.array(hist[k])

    print(f"    SEI thickness after {n_cycles} cycles: {hist['sei_thickness'][-1]*1e9:.2f} nm")
    print(f"    SEI capacity loss: {hist['cap_loss_SEI'][-1]*100:.3f}%")
    print(f"    DIS σ_vm peak: {hist['dis_vm_peak'][0]/1e6:.3f} MPa")
    n_cracks = int(hist['cracked'].sum())
    print(f"    SEI cracking events: {n_cracks}/{n_cycles} cycles")

    return hist
